define the feature functions¶

  • feature vector is defined by $(1, f_1(x, y), f_2(x, y), \cdots, f_{k-1}(x, y)) \in \mathbb{R}^k$
In [ ]:
def compute_feature1(point):

    # ++++++++++++++++++++++++++++++++++++++++++++++++++
    # complete the blanks
    #
    n = len(point)
    x = point[:,0].reshape(n,1)
    y = point[:,1].reshape(n,1)

    f1 = np.power(x,2)
    f2 = np.power(y,2)

    
    feature = np.concatenate((np.ones((n,1)),f1,f2),axis=1)


    #
    # ++++++++++++++++++++++++++++++++++++++++++++++++++

    return feature

def compute_feature2(point):

    # ++++++++++++++++++++++++++++++++++++++++++++++++++
    # complete the blanks
    #
    n = len(point)
    x = point[:,0].reshape(n,1)
    y = point[:,1].reshape(n,1)


    # f1 = np.sin(x)
    f2 = np.cos(y)
    f3 = np.power(x+0.5,3)+ np.power(x-1,3)
    f4 = np.power(y+0.5,3)+ np.power(y-1,3)

    feature = np.concatenate((np.ones((n,1)),f2,f3,f4),axis=1)
    #
    # ++++++++++++++++++++++++++++++++++++++++++++++++++

    return feature

define the linear regression function¶

  • $\theta = (\theta_0, \theta_1, \cdots, \theta_{k-1}) \in \mathbb{R}^k$
  • feature = $(1, f_1(x, y), \cdots, f_{k-1}(x, y)) \in \mathbb{R}^k$

def compute_linear_regression(theta, feature):

    # ++++++++++++++++++++++++++++++++++++++++++++++++++
    # complete the blanks
    #
    value = np.matmul(theta,feature.T)


    #
    # ++++++++++++++++++++++++++++++++++++++++++++++++++

    return value

def sigmoid(z):

    # ++++++++++++++++++++++++++++++++++++++++++++++++++
    # complete the blanks
    #
    value = np.divide(1,1+np.exp(-z))


    #
    # ++++++++++++++++++++++++++++++++++++++++++++++++++

    return value

define the logistic regression function¶

  • $\theta = (\theta_0, \theta_1, \cdots, \theta_{k-1}) \in \mathbb{R}^k$
  • feature $= (1, f_1(x, y), \cdots, f_{k-1}(x, y) \in \mathbb{R}^k$
In [ ]:
def compute_logistic_regression(theta, feature):

    # ++++++++++++++++++++++++++++++++++++++++++++++++++
    # complete the blanks
    #

    value = sigmoid(compute_linear_regression(theta, feature))


    #
    # ++++++++++++++++++++++++++++++++++++++++++++++++++

    return value

define the residual function¶

  • $\theta = (\theta_0, \theta_1, \cdots, \theta_{k-1}) \in \mathbb{R}^k$
  • feature $= (1, f_1(x, y), \cdots, f_{k-1}(x, y) \in \mathbb{R}^k$
  • label $= l \in \{0, 1\}^k$
In [ ]:
def compute_residual(theta, feature, label):

    # ++++++++++++++++++++++++++++++++++++++++++++++++++
    # complete the blanks
    #

    h = compute_logistic_regression(theta, feature)

    residual = -label*np.log(h) - (1-label)*np.log(1-h)


    #
    # ++++++++++++++++++++++++++++++++++++++++++++++++++

    return residual

define the loss function for the logistic regression¶

  • $\theta = (\theta_0, \theta_1, \cdots, \theta_{k-1}) \in \mathbb{R}^k$
  • feature $= (1, f_1(x, y), \cdots, f_{k-1}(x, y) \in \mathbb{R}^k$
  • label $= l \in \{0, 1\}^k$
In [ ]:
def compute_loss(theta, feature, label):

    # ++++++++++++++++++++++++++++++++++++++++++++++++++
    # complete the blanks
    #

    n = len(label)
    residual = compute_residual(theta, feature, label)
    loss = np.divide(np.sum(residual),n)


    #
    # ++++++++++++++++++++++++++++++++++++++++++++++++++

    return loss

define the gradient of the loss with respect to the model parameter $\theta$¶

  • $\theta = (\theta_0, \theta_1, \cdots, \theta_{k-1}) \in \mathbb{R}^k$
  • feature $= (1, f_1(x, y), \cdots, f_{k-1}(x, y) \in \mathbb{R}^k$
  • label $= l \in \{0, 1\}^k$
In [ ]:
def compute_gradient(theta, feature, label):

    # ++++++++++++++++++++++++++++++++++++++++++++++++++
    # complete the blanks
    #
    n= len(label)
    h = compute_logistic_regression(theta, feature)
    gradient = np.divide(np.matmul(feature.T,(h-label)),n)

    #
    # ++++++++++++++++++++++++++++++++++++++++++++++++++

    return gradient

compute the accuracy of the prediction for point with a given model parameter¶

In [ ]:
def compute_accuracy(theta, feature, label):

    # ++++++++++++++++++++++++++++++++++++++++++++++++++
    # complete the blanks
    #
    n = len(feature)
    threshold = 0.5

    h = compute_logistic_regression(theta, feature)

    prediction = np.where( h >= threshold, 1, 0)

    accuracy = np.divide(np.sum(label == prediction),n) 
    


    #
    # ++++++++++++++++++++++++++++++++++++++++++++++++++

    return accuracy
